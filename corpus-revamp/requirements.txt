# corpus-revamp Python deps. Install into a venv on a machine where YouTube downloads work.
#   python3 -m venv .venv && . .venv/bin/activate
#   pip install -r requirements.txt
#   # torch: for CPU-only boxes use the CPU wheel index (much smaller, no CUDA):
#   pip install torch --index-url https://download.pytorch.org/whl/cpu
#
# System binaries also required (install via apt/brew): ffmpeg, tesseract-ocr, nodejs, deno
# (node + deno are JS runtimes yt-dlp needs to solve YouTube's signature/n challenges).
# The YuNet face model is vendored in this dir.
#
# CRITICAL for downloads: modern YouTube hides downloadable formats behind a JS
# "EJS" challenge — without the solver yt-dlp returns ONLY storyboards ("Requested
# format is not available"). Installing yt-dlp-ejs (below) + having deno on PATH +
# FRESH full-auth cookies is what makes real 144p-1080p formats appear and download.
yt-dlp
yt-dlp-ejs              # EJS challenge solver — REQUIRED or only storyboards are returned
opencv-python-headless
pytesseract
numpy
faster-whisper          # transcript fallback when a video has no YouTube captions
torch                   # for the BLIP local VLM (use the CPU wheel index on CPU-only boxes)
transformers            # BLIP captioner
pillow
safetensors
